弹出框后退会返回到上一个弹出框的界面或者将上一个界面显示在弹出框中

以下为禁用弹出框后退页面的方法。

``.on("keydown", banBackSpace);``

```
function banBackSpace(event) {

    var elem = event.target || event.srcElement,

        nodeName = elem.nodeName.toLowerCase(),

        elemType = "

    if (event.keyCode == 8) {

        if (nodeName != 'input' && nodeName != 'textarea') {

            return false;

        } else if (nodeName == "input") {

            elemType = elem.type;

            if (elemType != 'text' && elemType != 'textarea' && elemType != 'password' && elemType != 'file' && (elem.readOnly == true || elem.disabled == true)) {

                return false;

            }

        }

    }

}
```
